#include<bits/stdc++.h>
using namespace std ;
vector<int> maxSlidingWindow(vector<int>& nums, int k) {
        int n = nums.size() ;
        deque<int> dq ;
        vector<int> ans(n - k + 1) ;
        int left = 0 ;
        for(int i = 0; i < n; i++){
            while(!dq.empty() && dq.front() < nums[i]){
                dq.pop_front() ;
            }
            while(!dq.empty() && dq.back() < nums[i]){
                dq.pop_back() ;
            }
            dq.push_back(nums[i]) ;
            if(k <= i + 1){
                ans[i + 1 - k] = dq.front() ;
                if(dq.front() == nums[left]) dq.pop_front() ;
                left++ ;
            }
        }
        return ans ;
    }
